## Coin Change
[LeetCode Link](https://leetcode.com/problems/coin-change/)
```cpp
int findMinCoins(int amount, const vector<int>& coins, vector<int>& dp) {
    if (amount < 0) return -1;  // Invalid amount
    if (amount == 0) return 0;  // No coins needed for amount 0
    if (dp[amount] != -1) return dp[amount];  // Return cached result

    int minCoins = INT_MAX;
    for (int coin : coins) {
        int res = findMinCoins(amount - coin, coins, dp);
        if (res != -1) {
            minCoins = min(minCoins, res + 1);
        }
    }

    dp[amount] = (minCoins == INT_MAX) ? -1 : minCoins;
    return dp[amount];
}

int coinChange(vector<int>& coins, int amount) {
    vector<int> dp(amount + 1, -1); 
    return findMinCoins(amount, coins, dp);
}
